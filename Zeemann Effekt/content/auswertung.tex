Energie \autoref{eq:anzeeman} ausgegangen:
\begin{equation*}
  \Delta E = g_{ij} \mu_B B m_J\\

\begin{equation*}
  \Delta E=E(\lambda+\delta \lambda) - E(\lambda)
\end{equation*}

\begin{equation*}

\end{equation*}

\begin{equation*}

\end{equation*}
